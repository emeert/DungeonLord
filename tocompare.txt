
#include <stdio.h>

#define WIDTH 20
#define HEIGHT 10

void clear_screen(void) {
    for (int i = 0; i < 50; i++) {
        printf("\n");
    }
}

void render(char map[HEIGHT][WIDTH], int player_x, int player_y) {
    clear_screen();

    for (int y = 0; y < HEIGHT; y++) {
        for (int x = 0; x < WIDTH; x++) {
            if (x == player_x && y == player_y) {
                printf("@");
            } else {
                printf("%c", map[y][x]);
            }
        }
        printf("\n");  // Add a newline at the end of each row
    }
}

int main() {
    char map[HEIGHT][WIDTH] = {
        "####################",
        "#.................#",
        "#.................#",
        "#.................#",
        "#.................#",
        "#.................#",
        "#.................#",
        "#.................#",
        "#.................#",
        "####################"
    };

    int player_x = 1, player_y = 1;
    char input;

    while (1) {
        render(map, player_x, player_y);
        printf("Enter move (w/a/s/d): ");
        input = getchar();
        getchar();  // Consume the newline character

        switch (input) {
            case 'w': player_y--; break;
            case 's': player_y++; break;
            case 'a': player_x--; break;
            case 'd': player_x++; break;
        }
    }

    return 0;
}
